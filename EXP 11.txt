// Parent class
class Vehicle {
 void start() {
 System.out.println("Vehicle is starting...");
 }
}
// Child class 1
class Car extends Vehicle {
 @Override
 void start() {
 System.out.println("Car starts with a key.");
 }
}
// Child class 2
class Bike extends Vehicle {
 @Override
 void start() {
 System.out.println("Bike starts with a kick or button.");
 }
}
// Child class 3
class Truck extends Vehicle {
 @Override
 void start() {
 System.out.println("Truck starts with a heavy engine sound.");
 }
}
// Main class
public class PolymorphismDemo1 {
 public static void main(String[] args) {
 // Polymorphism: Parent reference → Child objects
 Vehicle v;
 v = new Car();
 v.start(); // Calls Car’s start()
 v = new Bike();
 v.start(); // Calls Bike’s start()
 v = new Truck();
 v.start(); // Calls Truck’s start()
 }
}
